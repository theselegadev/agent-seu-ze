CREATE DATABASE seu_ze;
USE seu_ze;

CREATE TABLE barbeiro(
    id INT PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(255),
    endereco VARCHAR(255),
    telefone VARCHAR(20),
    senha VARCHAR(255)  
);

CREATE INDEX idx_barbeiro_nome ON barbeiro(nome);
CREATE INDEX idx_barbeiro_telefone ON barbeiro(telefone);

CREATE TABLE cliente(
    id INT PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(255),
    telefone VARCHAR(20),
    email VARCHAR(255),
    senha VARCHAR(255)
);

CREATE INDEX idx_cliente_telefone ON cliente(telefone);
CREATE INDEX idx_cliente_id ON cliente(id);

CREATE TABLE agenda(
  id INT PRIMARY KEY AUTO_INCREMENT,
  id_barbeiro INT,
  id_cliente INT,
  data DATETIME,
  FOREIGN KEY (id_barbeiro) REFERENCES barbeiro(id),
  FOREIGN KEY (id_cliente) REFERENCES cliente(id)
);

CREATE INDEX idx_agenda_barbeiro ON agenda(id_barbeiro);
CREATE INDEX idx_agenda_cliente ON agenda(id_cliente);
CREATE INDEX idx_agenda_data ON agenda(data);
